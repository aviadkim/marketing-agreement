if (submitButton) { submitButton.disabled = false; submitButton.innerHTML = 'סיים והגש טופס'; } return false; } } // Form validation functions function validateForm(form) { if (!form.checkValidity()) { form.reportValidity(); return false; } const requiredCheckboxes = [ 'riskAcknowledgement', 'independentDecision', 'updateCommitment', 'finalConfirmation' ]; const missingFields = requiredCheckboxes.filter(field => !form.elements[field].checked); if (missingFields.length > 0) { showMessage('יש לסמן את כל תיבות הסימון הנדרשות', 'error'); return false; } return true; } // Event listeners setup document.addEventListener('DOMContentLoaded', () => { const form = document.getElementById('section4-form'); const submitButton = document.getElementById('finalSubmit'); const signaturePad = window.signaturePad; if (form && submitButton) { // Handle form submission submitButton.addEventListener('click', async (e) => { e.preventDefault(); if (form.checkValidity() && !signaturePad.isEmpty()) { const buttonText = submitButton.innerHTML; submitButton.disabled = true; submitButton.innerHTML = ''; try { await submitSection4(); } catch (error) { console.error('Submission failed:', error); showMessage('שגיאה בשליחת הטופס', 'error'); } finally { submitButton.disabled = false; submitButton.innerHTML = buttonText; } } else { if (signaturePad.isEmpty()) { showMessage('נא להוסיף חתימה', 'error'); } else { showMessage('נא למלא את כל השדות הנדרשים', 'error'); } } }); // Handle back button const backButton = document.getElementById('btnBack'); if (backButton) { backButton.addEventListener('click', () => { window.location.href = '/sections/section3.html'; }); } // Load saved data if exists const savedData = localStorage.getItem('section4Data'); if (savedData) { try { const data = JSON.parse(savedData); Object.entries(data).forEach(([key, value]) => { const input = form.querySelector(`[name="${key}"]`); if (input && input.type === 'checkbox') { input.checked = value === 'on' || value === true; } }); // Restore signature if available if (data.signature) { signaturePad.fromDataURL(data.signature); document.getElementById('signatureData').value = data.signature;
